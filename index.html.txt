    <script>
        // When the user submits the form
        document.getElementById("passwordForm").addEventListener("submit", function(event) {
            event.preventDefault();  // Prevent the page from reloading

            const currentPassword = document.getElementById("currentPassword").value;
            const newPassword = document.getElementById("newPassword").value;
            const confirmPassword = document.getElementById("confirmPassword").value;

            // Check if the passwords match
            if (newPassword !== confirmPassword) {
                alert("Passwords do not match!");
                return;
            }

            // Store the data in localStorage
            localStorage.setItem("currentPassword", currentPassword);
            localStorage.setItem("newPassword", newPassword);

            alert("Password changed successfully!");
        });

        // Load data from localStorage when the page loads
        window.onload = function() {
            const storedCurrentPassword = localStorage.getItem("currentPassword");
            const storedNewPassword = localStorage.getItem("newPassword");

            if (storedCurrentPassword && storedNewPassword) {
                console.log("Data stored previously:");
                console.log("Current Password: " + storedCurrentPassword);
                console.log("New Password: " + storedNewPassword);
            }
        };
    </script>
